function renderMd(text) {
  return text.split(/(\*\*[^*]+\*\*)/).map((p,i)=>
    p.startsWith("**")&&p.endsWith("**")

}

              onClick={()=>{ if(window.innerWidth>1024){setSidebar(s=>!s);}else{setMobileSidebar(o=>!o);} }}
              style={{width:34,height:34,borderRadius:9,background:C.bgSurface,border:`1px solid ${C.line}`,display:"flex",alignItems:"center",justifyContent:"center",color:C.textMuted,cursor:"pointer",transition:"all .15s",flexShrink:0}}
              onMouseEnter={e=>{e.currentTarget.style.borderColor=C.brandBorder;e.currentTarget.style.color=C.brand;}}
              onMouseLeave={e=>{e.currentTarget.style.borderColor=C.line;e.currentTarget.style.color=C.textMuted;}}>
              {Ic.menu}
            </button>
            {activeDoc&&(
              <div className="dash-header-doc" style={{display:"flex",alignItems:"center",gap:8,padding:"4px 10px 4px 5px",borderRadius:10,background:"#fff",border:`1px solid ${C.line}`,boxShadow:C.shadow}}>
                <DocBadge color={activeDoc.color}/>
                <div>
                  <div style={{fontSize:12.5,fontWeight:700,color:C.text,whiteSpace:"nowrap",overflow:"hidden",maxWidth:"30vw",textOverflow:"ellipsis"}}>{activeDoc.name}</div>
                  <div style={{fontSize:11,color:C.textMuted}}>{activeDoc.pages}p · {activeDoc.size}</div>
                </div>
              </div>
            )}
          </div>
          <div className="dash-header-right" style={{display:"flex",alignItems:"center",gap:6,flexShrink:0}}>
            <div className="disclaimer-chip" style={{display:"flex",alignItems:"center",gap:5,padding:"5px 10px",borderRadius:20,background:C.accentLight,border:`1px solid ${C.accentBorder}`}}>
              <span style={{color:C.accent,display:"flex"}}>{Ic.info}</span>
              <span style={{fontSize:11,fontWeight:700,color:C.accent,whiteSpace:"nowrap"}}>Not legal advice</span>
            </div>
            <div style={{display:"flex",alignItems:"center",gap:5,padding:"5px 10px",borderRadius:20,background:C.greenLight,border:`1px solid rgba(5,150,105,.18)`,flexShrink:0}}>
              <div style={{width:6,height:6,borderRadius:"50%",background:C.green,animation:"pulse 2.5s infinite"}}/>
              <span style={{fontSize:11,fontWeight:700,color:C.green}}>AI Ready</span>
            </div>
          </div>
        </header>

        {/* Scrollable content */}
        <div style={{flex:1,overflowY:"auto"}}>
          {!hasAsked
            ?<WelcomeState onSend={send} docName={activeDoc?.name||"Your HOA Document"}/>
            :(
              <div className="msg-max" style={{maxWidth:800,margin:"0 auto",padding:"24px clamp(12px,3vw,24px)"}}>
                {msgs.map(msg=>(
                  <div key={msg.id} style={{marginBottom:22,display:"flex",flexDirection:msg.role==="user"?"row-reverse":"row",alignItems:"flex-start",gap:10,animation:"fadeUp .25s ease-out both"}}>
                    {msg.role==="ai"
                      ?<div style={{width:32,height:32,borderRadius:10,flexShrink:0,marginTop:2,background:`linear-gradient(145deg,${C.brand},${C.brandMid})`,display:"flex",alignItems:"center",justifyContent:"center",color:"#fff",boxShadow:`0 3px 10px rgba(27,67,50,.28)`}}>{Ic.ai}</div>
                      :<Avatar name="ZB" size={32}/>
                    }
                    <div style={{maxWidth:"min(78%,600px)"}}>
                      {msg.role==="ai"&&msg.tag&&<VerdictBanner tag={msg.tag}/>}
                      <div style={{padding:"12px 15px",background:msg.role==="user"?`linear-gradient(145deg,${C.brand},${C.brandMid})`:"#fff",borderRadius:msg.role==="user"?"16px 3px 16px 16px":"3px 16px 16px 16px",color:msg.role==="user"?"#fff":C.text,fontSize:14,lineHeight:1.7,boxShadow:msg.role==="user"?`0 4px 18px rgba(27,67,50,.26)`:C.shadow,border:msg.role==="ai"?`1px solid ${C.line}`:"none"}}>
                        {renderMd(msg.text)}
                      </div>
                      {msg.citations?.length>0&&(
                        <div style={{marginTop:7}}>
                          <div style={{fontSize:10,fontWeight:700,color:C.textFaint,letterSpacing:".08em",textTransform:"uppercase",marginBottom:4}}>Cited clauses</div>
                          {msg.citations.map((c,i)=>(
                            <div key={i} style={{display:"flex",alignItems:"center",gap:7,padding:"7px 10px",background:C.brandLight,borderRadius:9,border:`1px solid ${C.brandBorder}`,cursor:"pointer",fontSize:12,marginBottom:4,transition:"all .15s"}}
                              onMouseEnter={e=>e.currentTarget.style.background="rgba(27,67,50,.1)"}
                              onMouseLeave={e=>e.currentTarget.style.background=C.brandLight}>
                              <span className="mono" style={{background:C.brand,color:"#fff",borderRadius:5,padding:"2px 6px",fontSize:9.5,fontWeight:600,flexShrink:0,whiteSpace:"nowrap"}}>p.{c.page} {c.clause}</span>
                              <span style={{color:C.textMuted,lineHeight:1.5}}>{c.text}</span>
                            </div>
                          ))}
                        </div>
                      )}
                      {msg.role==="ai"&&msg.citations?.length>0&&(
                        <button onClick={()=>exportAnswer(msg)}
                          style={{marginTop:7,display:"inline-flex",alignItems:"center",gap:6,padding:"6px 12px",borderRadius:8,fontSize:12,fontWeight:700,color:C.textMuted,background:"#fff",border:`1px solid ${C.line}`,cursor:"pointer",transition:"all .15s"}}
                          onMouseEnter={e=>{e.currentTarget.style.borderColor=C.brand;e.currentTarget.style.color=C.brand;e.currentTarget.style.background=C.brandLight;}}
                          onMouseLeave={e=>{e.currentTarget.style.borderColor=C.line;e.currentTarget.style.color=C.textMuted;e.currentTarget.style.background="#fff";}}>
                          {Ic.save} Save answer
                        </button>
                      )}
                      <div style={{fontSize:10.5,color:C.textFaint,marginTop:5,textAlign:msg.role==="user"?"right":"left"}}>{msg.time}</div>
                    </div>
                  </div>
                ))}
                {loading&&(
                  <div style={{display:"flex",alignItems:"flex-start",gap:10,marginBottom:20}}>
                    <div style={{width:32,height:32,borderRadius:10,background:`linear-gradient(145deg,${C.brand},${C.brandMid})`,display:"flex",alignItems:"center",justifyContent:"center",color:"#fff",boxShadow:`0 3px 10px rgba(27,67,50,.28)`,flexShrink:0}}>{Ic.ai}</div>
                    <div style={{background:"#fff",borderRadius:"3px 16px 16px 16px",padding:"13px 16px",boxShadow:C.shadow,border:`1px solid ${C.line}`}}>
                      <div style={{display:"flex",gap:4}}>{[0,1,2].map(i=><div key={i} style={{width:7,height:7,borderRadius:"50%",background:C.brand,animation:`bounce 1.2s ${i*.16}s ease-in-out infinite`}}/>)}</div>
                    </div>
                  </div>
                )}
                <div ref={bottomRef}/>
              </div>
            )}
        </div>

        {/* Input bar */}
        <div className="input-bar-wrap" style={{padding:"0 clamp(12px,3vw,24px) 16px",maxWidth:800,margin:"0 auto",width:"100%"}}>
          <div style={{background:"#fff",borderRadius:16,padding:"9px 11px",boxShadow:C.shadowMd,border:`1px solid ${C.lineMid}`,display:"flex",alignItems:"flex-end",gap:7}}>
            <button style={{width:32,height:32,borderRadius:8,background:C.bgSurface,border:`1px solid ${C.line}`,color:C.textMuted,display:"flex",alignItems:"center",justifyContent:"center",flexShrink:0,cursor:"pointer",transition:"all .15s"}}
              onMouseEnter={e=>{e.currentTarget.style.borderColor=C.brandBorder;e.currentTarget.style.color=C.brand;}}
              onMouseLeave={e=>{e.currentTarget.style.borderColor=C.line;e.currentTarget.style.color=C.textMuted;}}>
              {Ic.attach}
            </button>
            <textarea ref={textRef} value={input} onChange={e=>setInput(e.target.value)}
              onKeyDown={e=>{if(e.key==="Enter"&&!e.shiftKey){e.preventDefault();send();}}}
              onInput={e=>{e.target.style.height="auto";e.target.style.height=Math.min(e.target.scrollHeight,120)+"px";}}
              placeholder="Ask about your HOA rules…" rows={1}
              style={{flex:1,border:"none",fontSize:14,color:C.text,background:"transparent",lineHeight:1.6,padding:"4px 0",maxHeight:120,overflowY:"auto",resize:"none",outline:"none"}}/>
            <button className="send-btn" onClick={()=>send()} disabled={!input.trim()}
              style={{width:34,height:34,borderRadius:9,flexShrink:0,background:input.trim()?`linear-gradient(145deg,${C.brand},${C.brandMid})`:C.bgSurface,color:input.trim()?"#fff":C.textFaint,display:"flex",alignItems:"center",justifyContent:"center",border:input.trim()?"none":`1px solid ${C.line}`,transition:"all .18s",boxShadow:input.trim()?`0 4px 14px rgba(27,67,50,.38)`:"none",cursor:input.trim()?"pointer":"not-allowed"}}>
              {Ic.send}
            </button>
          </div>
          <p style={{textAlign:"center",fontSize:11,color:C.textFaint,marginTop:6}}>HOA Helper provides information based on your uploaded document. This is not legal advice.</p>
        </div>
      </main>
    </div>
  );
}
